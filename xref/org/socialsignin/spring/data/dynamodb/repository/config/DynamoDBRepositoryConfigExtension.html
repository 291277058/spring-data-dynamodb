16  package org.socialsignin.spring.data.dynamodb.repository.config;
17  
18  import org.socialsignin.spring.data.dynamodb.mapping.DynamoDBMappingContext;
19  import org.socialsignin.spring.data.dynamodb.repository.support.DynamoDBRepositoryFactoryBean;
20  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
21  import org.springframework.core.annotation.AnnotationAttributes;
22  import org.springframework.data.config.ParsingUtils;
23  import org.springframework.data.repository.config.AnnotationRepositoryConfigurationSource;
24  import org.springframework.data.repository.config.RepositoryConfigurationExtensionSupport;
25  import org.springframework.data.repository.config.XmlRepositoryConfigurationSource;
26  import org.springframework.util.Assert;
27  import org.springframework.util.StringUtils;
28  import org.w3c.dom.Element;
29  
30  /**
31   * @author Michael Lavelle
32   * @author Sebastian Just
33   */
34  public class DynamoDBRepositoryConfigExtension extends RepositoryConfigurationExtensionSupport {
35  
36  	private static final String DEFAULT_AMAZON_DYNAMO_DB_BEAN_NAME = "amazonDynamoDB";
37  
38  	private static final String DYNAMO_DB_MAPPER_CONFIG_REF = "dynamodb-mapper-config-ref";
39  
40  	private static final String DYNAMO_DB_OPERATIONS_REF = "dynamodb-operations-ref";
41  
42  	private static final String AMAZON_DYNAMODB_REF = "amazon-dynamodb-ref";
43  
44  	private static final String MAPPING_CONTEXT_REF = "mapping-context-ref";
45  
46  	@Override
47  	public String getRepositoryFactoryBeanClassName() {
48  		return DynamoDBRepositoryFactoryBean.class.getName();
49  	}
50  
51  	@Override
52  	public void postProcess(BeanDefinitionBuilder builder, AnnotationRepositoryConfigurationSource config) {
53  		AnnotationAttributes attributes = config.getAttributes();
54  
55  		postProcess(builder, attributes.getString("amazonDynamoDBRef"), attributes.getString("dynamoDBMapperConfigRef"),
56  				attributes.getString("dynamoDBOperationsRef"), attributes.getString("mappingContextRef"));
57  
58  	}
59  
60  	/*
61  	 * (non-Javadoc)
62  	 * 
63  	 * @see org.springframework.data.repository.config.
64  	 * RepositoryConfigurationExtensionSupport
65  	 * #postProcess(org.springframework.beans
66  	 * .factory.support.BeanDefinitionBuilder, org.springframework.data.repository
67  	 * .config.XmlRepositoryConfigurationSource)
68  	 */
69  	@Override
70  	public void postProcess(BeanDefinitionBuilder builder, XmlRepositoryConfigurationSource config) {
71  
72  		Element element = config.getElement();
73  
74  		ParsingUtils.setPropertyReference(builder, element, AMAZON_DYNAMODB_REF, "amazonDynamoDB");
75  		ParsingUtils.setPropertyReference(builder, element, DYNAMO_DB_MAPPER_CONFIG_REF, "dynamoDBMapperConfig");
76  		ParsingUtils.setPropertyReference(builder, element, DYNAMO_DB_OPERATIONS_REF, "dynamoDBOperations");
77  		ParsingUtils.setPropertyReference(builder, element, MAPPING_CONTEXT_REF, "dynamoDBMappingContext");
78  
79  	}
80  
81  	private void postProcess(BeanDefinitionBuilder builder, String amazonDynamoDBRef, String dynamoDBMapperConfigRef,
82  			String dynamoDBOperationsRef, String dynamoDBMappingContextRef) {
83  
84  		if (StringUtils.hasText(dynamoDBOperationsRef)) {
85  			builder.addPropertyReference("dynamoDBOperations", dynamoDBOperationsRef);
86  			Assert.isTrue(!StringUtils.hasText(amazonDynamoDBRef),
87  					"Cannot specify both amazonDynamoDB bean and dynamoDBOperationsBean in repository configuration");
88  			Assert.isTrue(!StringUtils.hasText(dynamoDBMapperConfigRef),
89  					"Cannot specify both dynamoDBMapperConfigBean bean and dynamoDBOperationsBean in repository configuration");
90  		} else {
91  
92  			amazonDynamoDBRef = StringUtils.hasText(amazonDynamoDBRef)
93  					? amazonDynamoDBRef
94  					: DEFAULT_AMAZON_DYNAMO_DB_BEAN_NAME;
95  
96  			builder.addPropertyReference("amazonDynamoDB", amazonDynamoDBRef);
97  
98  			if (StringUtils.hasText(dynamoDBMapperConfigRef)) {
99  				builder.addPropertyReference("dynamoDBMapperConfig", dynamoDBMapperConfigRef);
100 			}
101 		}
102 
103 		if (StringUtils.hasText(dynamoDBMappingContextRef)) {
104 			builder.addPropertyReference("dynamoDBMappingContext", dynamoDBMappingContextRef);
105 		} else {
106 			builder.addPropertyValue("dynamoDBMappingContext", new DynamoDBMappingContext());
107 		}
108 	}
109 
110 	@Override
111 	protected String getModulePrefix() {
112 		return "dynamoDB";
113 	}
114 
115 }
