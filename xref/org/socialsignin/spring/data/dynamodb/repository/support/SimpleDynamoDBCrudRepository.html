37  /**
38   * Default implementation of the
39   * {@link org.springframework.data.repository.CrudRepository} interface.
40   * 
41   * @param <T>
42   *            the type of the entity to handle
43   * @param <ID>
44   *            the type of the entity's identifier
45   */
46  public class SimpleDynamoDBCrudRepository<T, ID>
47  		implements
48  			DynamoDBCrudRepository<T, ID>,
49  			SortHandler,
50  			ExceptionHandler {
51  
52  	protected DynamoDBEntityInformation<T, ID> entityInformation;
53  
54  	protected Class<T> domainType;
55  
56  	protected EnableScanPermissions enableScanPermissions;
57  
58  	protected DynamoDBOperations dynamoDBOperations;
59  
60  	public SimpleDynamoDBCrudRepository(DynamoDBEntityInformation<T, ID> entityInformation,
61  			DynamoDBOperations dynamoDBOperations, EnableScanPermissions enableScanPermissions) {
62  		Assert.notNull(entityInformation, "entityInformation must not be null");
63  		Assert.notNull(dynamoDBOperations, "dynamoDBOperations must not be null");
64  
65  		this.entityInformation = entityInformation;
66  		this.dynamoDBOperations = dynamoDBOperations;
67  		this.domainType = entityInformation.getJavaType();
68  		this.enableScanPermissions = enableScanPermissions;
69  	}
70  
71  	@Override
72  	public Optional<T> findById(ID id) {
73  
74  		Assert.notNull(id, "The given id must not be null!");
75  
76  		T result;
77  		if (entityInformation.isRangeKeyAware()) {
78  			result = dynamoDBOperations.load(domainType, entityInformation.getHashKey(id),
79  					entityInformation.getRangeKey(id));
80  		} else {
81  			result = dynamoDBOperations.load(domainType, entityInformation.getHashKey(id));
82  		}
83  
84  		return Optional.ofNullable(result);
85  	}
86  
87  	@Override
88  	public List<T> findAllById(Iterable<ID> ids) {
89  
90  		Assert.notNull(ids, "The given ids must not be null!");
91  
92  		// Works only with non-parallel streams!
93  		AtomicInteger idx = new AtomicInteger();
94  		List<KeyPair> keyPairs = StreamSupport.stream(ids.spliterator(), false).map(id -> {
95  
96  			Assert.notNull(id, "The given id  at position " + idx.getAndIncrement() + " must not be null!");
97  
98  			if (entityInformation.isRangeKeyAware()) {
99  				return new KeyPair().withHashKey(entityInformation.getHashKey(id))
100 						.withRangeKey(entityInformation.getRangeKey(id));
101 			} else {
102 				return new KeyPair().withHashKey(id);
103 			}
104 		}).collect(Collectors.toList());
105 
106 		Map<Class<?>, List<KeyPair>> keyPairsMap = Collections.<Class<?>, List<KeyPair>>singletonMap(domainType,
107 				keyPairs);
108 		return dynamoDBOperations.batchLoad(keyPairsMap);
109 	}

111 	@Override
112 	public <S extends T> S save(S entity) {
113 
114 		dynamoDBOperations.save(entity);
115 		return entity;
116 	}
117 
118 	/**
119 	 * {@inheritDoc}
120 	 * 
121 	 * @throws BatchWriteException
122 	 *             in case of an error during saving
123 	 */
124 	@Override
125 	public <S extends T> Iterable<S> saveAll(Iterable<S> entities)
126 			throws BatchWriteException, IllegalArgumentException {
127 
128 		Assert.notNull(entities, "The given Iterable of entities not be null!");
129 		List<FailedBatch> failedBatches = dynamoDBOperations.batchSave(entities);
130 
131 		if (failedBatches.isEmpty()) {
132 			// Happy path
133 			return entities;
134 		} else {
135 			// Error handling:
136 			throw repackageToException(failedBatches, BatchWriteException.class);
137 		}
138 	}

140 	@Override
141 	public boolean existsById(ID id) {
142 
143 		Assert.notNull(id, "The given id must not be null!");
144 		return findById(id).isPresent();
145 	}

147 	void assertScanEnabled(boolean scanEnabled, String methodName) {
148 		Assert.isTrue(scanEnabled, "Scanning for unpaginated " + methodName + "() queries is not enabled.  "
149 				+ "To enable, re-implement the " + methodName
150 				+ "() method in your repository interface and annotate with @EnableScan, or "
151 				+ "enable scanning for all repository methods by annotating your repository interface with @EnableScan");
152 	}

154 	@Override
155 	public List<T> findAll() {
156 
157 		assertScanEnabled(enableScanPermissions.isFindAllUnpaginatedScanEnabled(), "findAll");
158 		DynamoDBScanExpression scanExpression = new DynamoDBScanExpression();
159 		return dynamoDBOperations.scan(domainType, scanExpression);
160 	}
161 
162 	@Override
163 	public long count() {
164 		assertScanEnabled(enableScanPermissions.isCountUnpaginatedScanEnabled(), "count");
165 		final DynamoDBScanExpression scanExpression = new DynamoDBScanExpression();
166 		return dynamoDBOperations.count(domainType, scanExpression);
167 	}

169 	@Override
170 	public void deleteById(ID id) {
171 
172 		Assert.notNull(id, "The given id must not be null!");
173 
174 		Optional<T> entity = findById(id);
175 
176 		if (entity.isPresent()) {
177 			dynamoDBOperations.delete(entity.get());
178 
179 		} else {
180 			throw new EmptyResultDataAccessException(String.format("No %s entity with id %s exists!", domainType, id),
181 					1);
182 		}
183 	}
184 
185 	@Override
186 	public void delete(T entity) {
187 		Assert.notNull(entity, "The entity must not be null!");
188 		dynamoDBOperations.delete(entity);
189 	}
190 
191 	@Override
192 	public void deleteAll(Iterable<? extends T> entities) {
193 
194 		Assert.notNull(entities, "The given Iterable of entities not be null!");
195 		dynamoDBOperations.batchDelete(entities);
196 	}
197 
198 	@Override
199 	public void deleteAll() {
200 
201 		assertScanEnabled(enableScanPermissions.isDeleteAllUnpaginatedScanEnabled(), "deleteAll");
202 		dynamoDBOperations.batchDelete(findAll());
203 	}
